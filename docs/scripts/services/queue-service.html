angular.module('app').factory('Queue', function(){
  "use strict";

  var queue;


  class Queue {
    constructor(){
      this.fifo = [];  
      this.ready = true;
    }


    push(s){
      this.fifo.push(s);
    }

    pop(){
      return (this.fifo.length > 0 ? this.fifo.shift() : undefined);
    }

    peek(){
      return (this.fifo.length > 0 ? this.fifo[0] : undefined);
    }
  }
