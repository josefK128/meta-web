angular.module('app').factory('Keyframes', function(){
  "use strict";
  console.log("\nKeyframes service defined");

  var keyframes;
  var kfs = config.keyframes;


  class Keyframes {
    constructor(){}

    keyframe(name){
      return kfs[name] || {};
    }
